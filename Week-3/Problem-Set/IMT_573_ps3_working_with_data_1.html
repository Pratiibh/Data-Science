def get_season(date_string):
    month = int(date_string[5:7])
    season = "Autumn"
    if month in [12, 1, 2]:
        season = "Winter"
    elif month in [3, 4, 5]:
        season = "Spring"
    elif month in [6, 7, 8]:
        season = "Summer"
    return season
